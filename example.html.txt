<script>
const { createApp, computed } = Vue;

/* ---------- shared soul data ---------- */
const CX = 500, CY = 500, R = 330;
const ANGLES = { 9:-90, 1:-50, 2:-10, 3:30, 4:70, 5:110, 6:150, 7:190, 8:230 };
const RING = [9,1,2,3,4,5,6,7,8];

const SOULS = {
  1:{ words:['Integrity','Correction'],      color:'#8A99A8', triad:'BODY' },
  2:{ words:['Needed','Love'],               color:'#E0558C', triad:'HEART' },
  3:{ words:['Significance','Success'],      color:'#E8B53A', triad:'HEART' },
  4:{ words:['Identity','Compassion'],       color:'#9B59D0', triad:'HEART' },
  5:{ words:['Understanding','Observation'], color:'#4A6FE3', triad:'HEAD' },
  6:{ words:['Trust','Connection'],          color:'#2BA8A0', triad:'HEAD' },
  7:{ words:['Possibility','Freedom'],       color:'#F07F2D', triad:'HEAD' },
  8:{ words:['Power','Resolve'],             color:'#D04040', triad:'BODY' },
  9:{ words:['Ease','Harmony'],              color:'#7CB35C', triad:'BODY' },
};

const SIMPLE = [[1,7],[7,5],[5,8],[8,2],[2,4],[4,1],[3,6],[6,9],[9,3]];

const pos = id => {
  const a = ANGLES[id] * Math.PI / 180;
  return { x: CX + R*Math.cos(a), y: CY + R*Math.sin(a) };
};
const wingsOf  = id => { const i = RING.indexOf(id); return [RING[(i+8)%9], RING[(i+1)%9]]; };
const simpleOf = id => SIMPLE.find(p => p[0]===id)[1];
const hardOf   = id => SIMPLE.find(p => p[1]===id)[0];
const kinOf    = id => new Set([...wingsOf(id), simpleOf(id), hardOf(id)]);

/* ---------- SoulWheel component ---------- */
const SoulWheel = {
  template: '#soul-wheel-tpl',
  props: { selected: { type: Number, default: null }, showInfluence: { type: Boolean, default: false } },
  emits: ['select','reset'],
  setup(props){
    const cores = RING.map(id => ({ id, ...pos(id), ...SOULS[id] }));

    const wingArcs = RING.map((a,i) => {
      const b = RING[(i+1)%9], p1 = pos(a), p2 = pos(b);
      return { a, b, x1:p1.x, y1:p1.y, x2:p2.x, y2:p2.y,
               c1:SOULS[a].color, c2:SOULS[b].color,
               d:`M ${p1.x.toFixed(1)} ${p1.y.toFixed(1)} A ${R} ${R} 0 0 1 ${p2.x.toFixed(1)} ${p2.y.toFixed(1)}` };
    });

    const NODE_R = 78;
    const growthLines = SIMPLE.map(([a,b]) => {
      const p1 = pos(a), p2 = pos(b);
      const dx = p2.x-p1.x, dy = p2.y-p1.y, d = Math.hypot(dx,dy), ux = dx/d, uy = dy/d;
      return { a, b, color: SOULS[a].color,
               x1: p1.x+ux*(NODE_R+4), y1: p1.y+uy*(NODE_R+4),
               x2: p2.x-ux*(NODE_R+10), y2: p2.y-uy*(NODE_R+10) };
    });

    const sel = computed(() => props.selected);
    const kin = computed(() => sel.value ? kinOf(sel.value) : null);

    const nodeState = id => {
      if (sel.value === null) return '';
      if (id === sel.value)   return 'lit';
      return kin.value.has(id) ? 'kin' : 'dim';
    };
    const wingState = w => {
      if (sel.value === null) return '';
      return (w.a === sel.value || w.b === sel.value) ? 'active' : 'faded';
    };
    const growthState = g => {
      if (sel.value === null) return '';
      return (g.a === sel.value || g.b === sel.value) ? 'active' : 'faded';
    };

    /* outer influence: from center(0) THROUGH each core to OUTSIDE (ghost-dashed, matches core color) */
    const INF_EXTEND = 260;
    const influenceLines = computed(() => {
      if (!props.showInfluence) return [];
      return RING.map(id => {
        const p = pos(id);
        const vx = p.x - CX, vy = p.y - CY;
        const vd = Math.hypot(vx, vy);
        const ux = vx / vd, uy = vy / vd;
        return {
          id,
          color: SOULS[id].color,
          x2: p.x + ux * INF_EXTEND,
          y2: p.y + uy * INF_EXTEND
        };
      });
    });

    const roleTags = computed(() => RING.map(id => {
      const p = pos(id);
      let label = '', show = false;
      if (sel.value !== null && id !== sel.value && kinOf(sel.value).has(id)){
        show = true;
        if (wingsOf(sel.value).includes(id)) label = 'wing';
        else if (simpleOf(sel.value) === id) label = 'simple';
        else label = 'hard';
      }
      const vx = p.x - CX, vy = p.y - CY, vd = Math.hypot(vx,vy);
      return { id, label, show, x: p.x + (vx/vd)*118, y: p.y + (vy/vd)*118 + 5 };
    }));

    return { cores, wingArcs, growthLines, influenceLines, nodeState, wingState, growthState, roleTags, CX, CY };
  }
};

/* ---------- page shell ---------- */
createApp({
  components: { SoulWheel },
  data(){ return { selected: null, showInfluence: false }; },
  computed: {
    current(){ return this.selected ? { id:this.selected, ...SOULS[this.selected] } : null; },
    currentRelations(){
      if (!this.selected) return [];
      const rel = (role,id) => ({ role, id, color:SOULS[id].color, words:SOULS[id].words.join(', ') });
      const [w1,w2] = wingsOf(this.selected);
      return [ rel('WING',w1), rel('WING',w2),
               rel('SIMPLE GROWTH', simpleOf(this.selected)),
               rel('HARD GROWTH',   hardOf(this.selected)) ];
    },
    soulBlocks(){
      if (!this.selected) return [{code:null, raw:'', copied:false}, {code:null, raw:'', copied:false}];
      
      const [w1, w2] = wingsOf(this.selected);
      const simpleT = simpleOf(this.selected);
      const hardT = hardOf(this.selected);
      
      const fmtArr = (id) => `["${SOULS[id].words[0]}","${SOULS[id].words[1]}"]`;
      const htmlArr = (id) => `<span class="punct">[</span><span class="str">"${SOULS[id].words[0]}"</span>,<span class="str">"${SOULS[id].words[1]}"</span><span class="punct">]</span>`;
      
      const makeBlock = (surfaceId, depthId) => {
        const raw = `SOUL:\n{\n  DRIVE: ${fmtArr(this.selected)},\n  SURFACE: ${fmtArr(surfaceId)},\n  DEPTH: ${fmtArr(depthId)},\n  GROWTH:\n  {\n    SIMPLE: ${fmtArr(simpleT)},\n    HARD: ${fmtArr(hardT)}\n  }\n}`;
        
        const code = `<span class="key">SOUL</span><span class="punct">:</span>\n<span class="brace">{</span>\n  <span class="key">DRIVE</span><span class="punct">:</span> ${htmlArr(this.selected)},\n  <span class="key">SURFACE</span><span class="punct">:</span> ${htmlArr(surfaceId)},\n  <span class="key">DEPTH</span><span class="punct">:</span> ${htmlArr(depthId)},\n  <span class="key">GROWTH</span><span class="punct">:</span>\n  <span class="brace">{</span>\n    <span class="key">SIMPLE</span><span class="punct">:</span> ${htmlArr(simpleT)},\n    <span class="key">HARD</span><span class="punct">:</span> ${htmlArr(hardT)}\n  <span class="brace">}</span>\n<span class="brace">}</span>`;
        
        return { code, raw, copied: false, surfaceId };
      };
      
      return [makeBlock(w1, w2), makeBlock(w2, w1)];
    }
  },
  methods: {
    onSelect(id){ this.selected = id; },
    onReset(){ this.selected = null; },
    async copyCode(block){
      try {
        await navigator.clipboard.writeText(block.raw);
        block.copied = true;
        setTimeout(() => { block.copied = false; }, 2000);
      } catch(e) {
        const ta = document.createElement('textarea');
        ta.value = block.raw;
        document.body.appendChild(ta);
        ta.select();
        document.execCommand('copy');
        document.body.removeChild(ta);
        block.copied = true;
        setTimeout(() => { block.copied = false; }, 2000);
      }
    }
  }
}).mount('#app');
</script>
